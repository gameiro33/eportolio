Group Project

A digital forensics agent.


	; Alice asks Bob about the available stock of 50-inch televisions
	(tell
	:sender Alice
	:receiver Bob
	:language KIF
	:ontology electronics-inventory
	:content
		(ask-one
		:sender Alice
		:receiver Bob
		:content
			(available-stock (television (size 50)))
		)
	)

	; Bob responds with the available stock
	(reply
	:sender Bob
	:receiver Alice
	:language KIF
	:ontology electronics-inventory
	:content
		(= (available-stock (television (size 50))) 25)
	)

	; Alice asks Bob about the number of HDMI slots on the 50-inch televisions
	(tell
	:sender Alice
	:receiver Bob
	:language KIF
	:ontology electronics-inventory
	:content
		(ask-one
		:sender Alice
		:receiver Bob
		:content
			(hdmi-slot-count (television (size 50)))
		)
	)

	; Bob responds with the HDMI slot count
	(reply
	:sender Bob
	:receiver Alice
	:language KIF
	:ontology electronics-inventory
	:content
		(= (hdmi-slot-count (television (size 50))) 4)
	)

	; Alice confirms receipt of information
	(tell
	:sender Alice
	:receiver Bob
	:language KIF
	:ontology electronics-inventory
	:content
		(acknowledge
		:sender Alice
		:receiver Bob
		:content "Information received. Thank you."
		)
	)
